leetcode-47.Permutations II-m

题目链接
Description
Given a collection of numbers that might contain duplicates, return all possible unique permutations.

Example:

Input: [1,1,2]
Output:
[
[1,1,2],
[1,2,1],
[2,1,1]
]

解题思路
与上一题求全排列的区别在于给出的数据有重复。对待无重复的全排列，给出的思路是：考虑递归的思想。将数组中的每一个数拿出来作为第一个数，将余下的n-1个数进行全排列，依次递归得到的所有排列即为全部数据的全排列。本题采用相同的思路，区别在于，每次进入递归函数时，先对数组进行排序，判断只将不同的数拿出来做为第一个数，对重复的数不再处理。

示例代码

void permutestep(vector<vector<int>>& res, vector<int> permu, int index)
{
sort(permu.begin()+index, permu.end());
if (permu.size() == index)
{
    res.push_back(permu);
}
else
{
    for (int i = 0; i < permu.size() - index; i++)
    {
        if ((i >= 1 && permu[index + i] == permu[index + i - 1]))
            continue;
        else
        {
            swap(permu[index], permu[index + i]);
            permutestep(res, permu, index + 1);
            swap(permu[index], permu[index + i]);
        }
    }
}
}

vector<vector<int>> permuteUnique(vector<int>& nums) {
sort(nums.begin(), nums.end());
vector<vector<int>> res;
permutestep(res, nums, 0);
for (int i = 0; i < res.size(); i++)
{
    for (int j = 0; j < nums.size(); j++)
    {
        cout << res[i][j];
    }
    cout << endl;
}
return res;
}
