leetcode-51/52.N-Queens-h

题目链接
Description
The n-queens puzzle is the problem of placing n queens on an n×n chessboard such that no two queens attack each other.

Given an integer n, return all distinct solutions to the n-queens puzzle.

Each solution contains a distinct board configuration of the n-queens’ placement, where ‘Q’ and ‘.’ both indicate a queen and an empty space respectively.

Example:
Input: 4
Output: [
[“.Q..”, // Solution 1
“…Q”,
“Q…”,
“..Q.”],

[“..Q.”, // Solution 2
“Q…”,
“…Q”,
“.Q..”]
]

解题思路
N皇后问题，每行每列和斜对角线都只能有一个皇后。深搜+剪枝的思路，按层搜索，在每层上搜索每一个位置，若满足条件，则保存状态递推搜索下一层。不满足条件则不再搜索。搜索到最后一层时，保存可行解。

示例代码

bool check(vector<string> map, int i, int j, int n)
{
    for (int k = 0; k < i; k++)
        if (map[k][j] == 'Q')
            return false;
    for (int k = i - 1, p = j - 1; k >= 0&&p >= 0; k--, p--)
        if (map[k][p] == 'Q')
            return false;
    for (int k = i - 1, p = j + 1; k >= 0&&p < n; k--, p++)
        if (map[k][p] == 'Q')
            return false;
    return true;
}
void dfs(vector<vector<string>> &res, vector<string> map, int i, int n)
{
    for (int j = 0; j < n; j++)
    {
        if (check(map, i, j, n))
        {
            if (i == n - 1)
            {
                map[i][j] = 'Q';
                res.push_back(map);
            }
            else
            {
                map[i][j] = 'Q';
                dfs(res, map, i + 1, n);
                map[i][j] = '.';
            }
        }
    }
}
vector<vector<string>> solveNQueens(int n) {
    vector<string> map;
    vector<vector<string>> res;
    for (int i = 0; i < n; i++)
    {
        string tmp="";
        for (int j = 0; j < n; j++)
            tmp += ".";
        map.push_back(tmp);
    }
    dfs(res,map, 0, n);
    for (int i = 0; i < res.size(); i++)
    {
        for (int j = 0; j < n; j++)
            cout << res[i][j] << endl;
        cout << endl;
    }
    return res;
}
